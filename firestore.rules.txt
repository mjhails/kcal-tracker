// Paste this into: Firebase console > Firestore Database > Rules tab > Publish
//
// What this does:
// - Each signed-in user can only read/write their OWN data under users/{uid}/...
// - Any signed-in user of this app can read/write the shared library
//   (combos + personal food entries you've both added) — that's the
//   "shared meal library" feature, deliberately open between the two of you.
// - Nobody who isn't signed in can read or write anything.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

      match /days/{date} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }

    match /shared/library {
      allow read, write: if request.auth != null;
    }
  }
}
